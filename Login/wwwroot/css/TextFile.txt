              public IActionResult Entrar(string returnUrl)
        {
            if (HttpContext.User.Claims.Count() >= 1)
            {
                if (string.IsNullOrEmpty(returnUrl))
                {
                    return Redirect("/");
                }
                else
                {
                    return Redirect(returnUrl);
                }
            }

            if (string.IsNullOrEmpty(returnUrl))
            {
                ViewData["ReturnUrl"] = "/";
            }
            else
            {
                ViewData["ReturnUrl"] = returnUrl;
            }

            return View(new Usuario());
        }
